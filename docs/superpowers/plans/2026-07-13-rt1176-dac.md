```c
// RT1176 12-bit DAC (DAC12) — true analog voltage output on TP18 (EVKB).
// From-SDK bring-up (fsl_dac12 + RM ch.89): the RT1062/Teensy 4 has no DAC,
// so the API shape follows Teensy 3.x analogWriteDAC0().
//
// RM 89.1 chip-specifics: VREFH is the only reference (DACRFS must be 1),
// no HW trigger, PTAT current ref unavailable -> internal (CR2.IREF).
// A current reference MUST be selected or the DAC silently doesn't convert.
// Non-FIFO mode: every DATA write converts immediately (no trigger).

#include "imxrt1176.h"
#include "core_pins.h"

static uint8_t dac_inited = 0;

static void dac_init(void)
{
	CCM_LPCG57_DIRECT = 1u;                             // ungate DAC clock
	DAC_CR  = DAC_CR_DACRFS_MASK | DAC_CR_TRGSEL_MASK;  // VREFH ref, SW trigger, FIFO off
	DAC_CR2 = DAC_CR2_IREF_MASK | DAC_CR2_OEN_MASK;     // internal current ref, buffer bypassed
	DAC_CR |= DAC_CR_DACEN_MASK;
	dac_inited = 1;
}

void analogWriteDAC0(uint32_t value)
{
	if (!dac_inited) dac_init();
	uint32_t bits = analogWriteResBits();   // Teensy 3.x semantics: scale the
	if (bits < 12u) {                       // current analogWriteResolution to
		value <<= (12u - bits);             // the DAC's native 12 bits
	} else if (bits > 12u) {
		value >>= (bits - 12u);
	}
	if (value > DAC_DATA_DATA0_MASK) value = DAC_DATA_DATA0_MASK;
	DAC_DATA = value;                       // non-FIFO: converts immediately
}
```

```cpp
#include "Arduino.h"
#include "core_pins.h"
#include "HardwareSerial.h"
#include "imxrt1176.h"
#include <math.h>

// QEMU gate (setup): asserts ONLY architecturally readable state -- CR config
// bits, FIFO pointers (PTR), status flags, watermark IRQ.  DATA is write-only
// on silicon, so the QEMU model's conversion codes are checked by the runner
// via the .dbg log, never by reading DATA back (anti-circularity rule).
// HW phase (loop): staircase + sine on TP18 for scope/DMM verification.

static volatile uint8_t dac_irq_fired = 0;
static void dac_isr(void) {
    // WMF is level status; mask the interrupt to stop refires. RMW of CR is
    // safe here: SWTRG/FIFORST/SWRST read as 0, but avoid clearing W1C flags.
    DAC_CR = (DAC_CR & ~(DAC_CR_UDFF_MASK | DAC_CR_OVFF_MASK)) & ~DAC_CR_WTMIE_MASK;
    dac_irq_fired = 1;
}

static void check(const char *name, uint32_t got, uint32_t want) {
    Serial1.print(name); Serial1.print("="); Serial1.print(got);
    if (got == want) Serial1.println(" OK");
    else { Serial1.print(" WANT "); Serial1.println(want); }
}

void setup() {
    Serial1.begin(115200);
    Serial1.println("RT1176 DAC12 test");

    // --- Phase 1: analogWriteDAC0 path ---
    analogWriteDAC0(128);                    // default 8-bit res -> code 0x800
    check("cr_dacen",  (DAC_CR & DAC_CR_DACEN_MASK)  ? 1u : 0u, 1);
    check("cr_dacrfs", (DAC_CR & DAC_CR_DACRFS_MASK) ? 1u : 0u, 1);
    check("cr_fifoen", (DAC_CR & DAC_CR_FIFOEN_MASK) ? 1u : 0u, 0);
    analogWriteResolution(12);
    analogWriteDAC0(4095);                   // -> 0xfff (dbg trace)
    analogWriteDAC0(2048);                   // -> 0x800 (dbg trace)

    // --- Phase 2: raw FIFO path (architectural readback; runs on HW too) ---
    DAC_CR &= ~DAC_CR_DACEN_MASK;
    DAC_CR |= DAC_CR_FIFORST_MASK;           // reset pointers (self-clearing)
    DAC_CR = DAC_CR_DACRFS_MASK | DAC_CR_TRGSEL_MASK | DAC_CR_FIFOEN_MASK
           | (4u << DAC_CR_WML_SHIFT);       // SW trigger, FIFO on, watermark 4
    DAC_CR |= DAC_CR_DACEN_MASK;
    DAC_DATA = 0x100; DAC_DATA = 0x200; DAC_DATA = 0x300; DAC_DATA = 0x400;
    check("wfp",  DAC_PTR & DAC_PTR_DACWFP_MASK, 4);
    check("rfp0", (DAC_PTR & DAC_PTR_DACRFP_MASK) >> DAC_PTR_DACRFP_SHIFT, 0);
    DAC_CR |= DAC_CR_SWTRG_MASK;             // pop one -> 3 left (< WML 4)
    check("rfp1", (DAC_PTR & DAC_PTR_DACRFP_MASK) >> DAC_PTR_DACRFP_SHIFT, 1);

    dac_irq_fired = 0;
    attachInterruptVector(IRQ_DAC, dac_isr);
    NVIC_ENABLE_IRQ(IRQ_DAC);
    DAC_CR |= DAC_CR_WTMIE_MASK;             // remaining 3 < WML 4 -> WMF -> IRQ
    for (uint32_t t = 0; t < 1000000u && !dac_irq_fired; t++) __asm__ volatile("nop");
    check("irq", dac_irq_fired, 1);
    check("wmf", (DAC_CR & DAC_CR_WMF_MASK) ? 1u : 0u, 1);
    DAC_CR |= DAC_CR_SWTRG_MASK;             // 2 left
    DAC_CR |= DAC_CR_SWTRG_MASK;             // 1 left -> nearly empty
    check("nemptf", (DAC_CR & DAC_CR_NEMPTF_MASK) ? 1u : 0u, 1);
    Serial1.println("[dac] done");

    // Restore the analogWriteDAC0 (non-FIFO) config for the HW phase in loop().
    DAC_CR &= ~DAC_CR_DACEN_MASK;
    DAC_CR = DAC_CR_DACRFS_MASK | DAC_CR_TRGSEL_MASK;
    DAC_CR |= DAC_CR_DACEN_MASK;
    analogWriteResolution(12);
}

// HW phase (scope/DMM on TP18; VREFH=1.8V nominal): 5-step staircase 2s each
// (0V, ~0.45V, ~0.9V, ~1.35V, ~1.8V), then 5s of 100Hz sine, repeating.
void loop() {
    static const uint16_t steps[] = {0, 1024, 2048, 3072, 4095};
    for (unsigned i = 0; i < 5; i++) {
        Serial1.print("dac="); Serial1.println(steps[i]);
        analogWriteDAC0(steps[i]);
        delay(2000);
    }
    Serial1.println("sine 100Hz x5s");
    uint32_t t0 = millis();
    while (millis() - t0 < 5000u) {
        float ph = (float)(micros() % 10000u) / 10000.0f;   // 100 Hz
        analogWriteDAC0((uint32_t)(2048.0f + 2047.0f * sinf(6.2831853f * ph)));
        delayMicroseconds(100);
    }
}
```

```c
#ifndef HW_MISC_IMXRT_DAC_H
#define HW_MISC_IMXRT_DAC_H

#include "hw/core/sysbus.h"
#include "hw/core/irq.h"
#include "qom/object.h"

#define TYPE_IMXRT_DAC "imxrt.dac"
OBJECT_DECLARE_SIMPLE_TYPE(IMXRTDACState, IMXRT_DAC)

#define IMXRT_DAC_MMIO_SIZE   0x4000
#define IMXRT_DAC_FIFO_DEPTH  16    /* PARAM.FIFOSZ = 0b011 on RT1170 (RM ch.89) */

/* Register offsets (PERI_DAC.h). */
#define DAC_VERID  0x00   /* RO */
#define DAC_PARAM  0x04   /* RO */
#define DAC_DATA   0x08   /* WO */
#define DAC_CR     0x0C
#define DAC_PTR    0x10   /* RO */
#define DAC_CR2    0x14

/* CR bits. */
#define DAC_CR_FULLF     (1u << 0)    /* ro */
#define DAC_CR_NEMPTF    (1u << 1)    /* ro */
#define DAC_CR_WMF       (1u << 2)    /* ro */
#define DAC_CR_UDFF      (1u << 3)    /* w1c */
#define DAC_CR_OVFF      (1u << 4)    /* w1c */
#define DAC_CR_FULLIE    (1u << 8)
#define DAC_CR_EMPTIE    (1u << 9)
#define DAC_CR_WTMIE     (1u << 10)
#define DAC_CR_SWTRG     (1u << 12)   /* self-clearing */
#define DAC_CR_TRGSEL    (1u << 13)
#define DAC_CR_DACRFS    (1u << 14)
#define DAC_CR_DACEN     (1u << 15)
#define DAC_CR_FIFOEN    (1u << 16)
#define DAC_CR_SWMD      (1u << 17)
#define DAC_CR_UVIE      (1u << 18)
#define DAC_CR_FIFORST   (1u << 21)   /* self-clearing */
#define DAC_CR_SWRST     (1u << 22)   /* self-clearing */
#define DAC_CR_DMAEN     (1u << 23)
#define DAC_CR_WML_SHIFT 24
#define DAC_CR_WML_MASK  0xFF000000u
/* Bits firmware may set and read back (flags/pulses are computed/handled). */
#define DAC_CR_STORED_MASK (DAC_CR_FULLIE | DAC_CR_EMPTIE | DAC_CR_WTMIE | \
                            DAC_CR_TRGSEL | DAC_CR_DACRFS | DAC_CR_DACEN | \
                            DAC_CR_FIFOEN | DAC_CR_SWMD | DAC_CR_UVIE |    \
                            DAC_CR_DMAEN | DAC_CR_WML_MASK)

#define DAC_DATA_MASK    0xFFFu

/* Read-only ID registers (RM reset values; VERID not consulted by firmware). */
#define DAC_VERID_VALUE  0x01000000u
#define DAC_PARAM_VALUE  0x00000003u  /* FIFOSZ=0b011: depth 16 */

struct IMXRTDACState {
    /*< private >*/
    SysBusDevice parent_obj;

    /*< public >*/
    MemoryRegion iomem;
    qemu_irq irq;

    uint32_t cr;        /* stored (non-flag) CR bits */
    uint32_t cr2;
    uint32_t fifo[IMXRT_DAC_FIFO_DEPTH];
    uint32_t wfp;       /* write pointer (mod depth) */
    uint32_t rfp;       /* read pointer (mod depth)  */
    uint32_t count;     /* FIFO fill level           */
    uint32_t output;    /* last converted 12-bit code */
    bool udff;
    bool ovff;
};

#endif /* HW_MISC_IMXRT_DAC_H */
```

```c
/*
 * NXP i.MX RT1170 DAC12 (12-bit digital-to-analog converter)
 *
 * Copyright (c) 2026 Nicholas Newdigate
 *
 * This program is free software; you can redistribute it and/or modify
 * it under the terms of the GNU General Public License version 2 or
 * (at your option) any later version.
 *
 * SPDX-License-Identifier: GPL-2.0-or-later
 *
 * Models the DAC12 digital interface: the one-level buffer (FIFO disabled --
 * any DATA write converts immediately) and the 16-deep FIFO with software
 * trigger (SWTRG pops the read pointer), watermark/nearly-empty/full status
 * and their interrupts, and the FIFO pointer readback in PTR.  There is no
 * analog back-end: a "conversion" stores the 12-bit code in s->output and
 * emits "imxrt_dac: convert 0x%03x" into the qemu log (-D), which the gate
 * runner greps -- DATA is write-only on real silicon, so firmware-visible
 * readback is limited to CR flags and PTR, exactly as modelled here.
 *
 * RT1170 chip-specifics (RM 89.1): no HW trigger option and no ITRM register.
 * CR2 is storage-only (current reference / buffer mode have no digital
 * side-effects).  DMA requests (CR.DMAEN) are not modelled (v1).
 */

#include "qemu/osdep.h"
#include "qemu/module.h"
#include "qemu/log.h"
#include "migration/vmstate.h"
#include "hw/misc/imxrt_dac.h"

static uint32_t imxrt_dac_wml(IMXRTDACState *s)
{
    return (s->cr & DAC_CR_WML_MASK) >> DAC_CR_WML_SHIFT;
}

static uint32_t imxrt_dac_flags(IMXRTDACState *s)
{
    uint32_t fl = 0;

    if (s->cr & DAC_CR_FIFOEN) {
        if (s->count == IMXRT_DAC_FIFO_DEPTH) {
            fl |= DAC_CR_FULLF;
        }
        if (s->count == 1) {
            fl |= DAC_CR_NEMPTF;
        }
        if (s->count < imxrt_dac_wml(s)) {
            fl |= DAC_CR_WMF;
        }
    }
    if (s->udff) {
        fl |= DAC_CR_UDFF;
    }
    if (s->ovff) {
        fl |= DAC_CR_OVFF;
    }
    return fl;
}

static void imxrt_dac_update_irq(IMXRTDACState *s)
{
    uint32_t fl = imxrt_dac_flags(s);
    bool level = ((s->cr & DAC_CR_WTMIE) && (fl & DAC_CR_WMF)) ||
                 ((s->cr & DAC_CR_EMPTIE) && (fl & DAC_CR_NEMPTF)) ||
                 ((s->cr & DAC_CR_FULLIE) && (fl & DAC_CR_FULLF)) ||
                 ((s->cr & DAC_CR_UVIE) && (fl & (DAC_CR_UDFF | DAC_CR_OVFF)));

    qemu_set_irq(s->irq, level);
}

static void imxrt_dac_convert(IMXRTDACState *s, uint32_t code)
{
    s->output = code & DAC_DATA_MASK;
    qemu_log("imxrt_dac: convert 0x%03x\n", s->output);
}

static void imxrt_dac_write_data(IMXRTDACState *s, uint32_t value)
{
    value &= DAC_DATA_MASK;

    if (!(s->cr & DAC_CR_DACEN)) {
        return;                     /* converter disabled: drop */
    }
    if (s->cr & DAC_CR_FIFOEN) {
        if (s->count < IMXRT_DAC_FIFO_DEPTH) {
            s->fifo[s->wfp] = value;
            s->wfp = (s->wfp + 1) % IMXRT_DAC_FIFO_DEPTH;
            s->count++;
        } else {
            s->ovff = true;         /* FIFO overflow */
        }
    } else {
        imxrt_dac_convert(s, value);    /* one-level buffer: immediate */
    }
    imxrt_dac_update_irq(s);
}

static void imxrt_dac_trigger(IMXRTDACState *s)
{
    if (!(s->cr & DAC_CR_DACEN) || !(s->cr & DAC_CR_FIFOEN)) {
        return;
    }
    if (s->count == 0) {
        s->udff = true;             /* trigger underflow */
    } else {
        imxrt_dac_convert(s, s->fifo[s->rfp]);
        s->rfp = (s->rfp + 1) % IMXRT_DAC_FIFO_DEPTH;
        s->count--;
    }
    imxrt_dac_update_irq(s);
}

static void imxrt_dac_reset_fifo(IMXRTDACState *s)
{
    s->wfp = 0;
    s->rfp = 0;
    s->count = 0;
}

static void imxrt_dac_reset_block(IMXRTDACState *s)
{
    s->cr = 0;
    s->cr2 = 0;
    imxrt_dac_reset_fifo(s);
    s->output = 0;
    s->udff = false;
    s->ovff = false;
}

static uint64_t imxrt_dac_read(void *opaque, hwaddr addr, unsigned int size)
{
    IMXRTDACState *s = IMXRT_DAC(opaque);

    switch (addr) {
    case DAC_VERID:
        return DAC_VERID_VALUE;
    case DAC_PARAM:
        return DAC_PARAM_VALUE;
    case DAC_DATA:
        return 0;                   /* write-only on silicon */
    case DAC_CR:
        return s->cr | imxrt_dac_flags(s);
    case DAC_PTR:
        return (s->rfp << 16) | s->wfp;
    case DAC_CR2:
        return s->cr2;
    default:
        qemu_log_mask(LOG_GUEST_ERROR,
                      "%s: bad read offset 0x%" HWADDR_PRIx "\n",
                      __func__, addr);
        return 0;
    }
}

static void imxrt_dac_write(void *opaque, hwaddr addr,
                            uint64_t value, unsigned int size)
{
    IMXRTDACState *s = IMXRT_DAC(opaque);

    switch (addr) {
    case DAC_DATA:
        imxrt_dac_write_data(s, value);
        return;
    case DAC_CR:
        if (value & DAC_CR_SWRST) {
            imxrt_dac_reset_block(s);
            imxrt_dac_update_irq(s);
            return;
        }
        if (value & DAC_CR_FIFORST) {
            imxrt_dac_reset_fifo(s);
        }
        if (value & DAC_CR_UDFF) {      /* write-1-clear */
            s->udff = false;
        }
        if (value & DAC_CR_OVFF) {
            s->ovff = false;
        }
        s->cr = value & DAC_CR_STORED_MASK;
        if (value & DAC_CR_SWTRG) {     /* self-clearing */
            imxrt_dac_trigger(s);
        }
        imxrt_dac_update_irq(s);
        return;
    case DAC_CR2:
        s->cr2 = value;
        return;
    case DAC_VERID:
    case DAC_PARAM:
    case DAC_PTR:
        return;                         /* read-only */
    default:
        qemu_log_mask(LOG_GUEST_ERROR,
                      "%s: bad write offset 0x%" HWADDR_PRIx "\n",
                      __func__, addr);
    }
}

static const MemoryRegionOps imxrt_dac_ops = {
    .read = imxrt_dac_read,
    .write = imxrt_dac_write,
    .endianness = DEVICE_LITTLE_ENDIAN,
    .impl = {
        .min_access_size = 4,
        .max_access_size = 4,
        .unaligned = false,
    },
};

static void imxrt_dac_reset_hold(Object *obj, ResetType type)
{
    IMXRTDACState *s = IMXRT_DAC(obj);

    imxrt_dac_reset_block(s);
    memset(s->fifo, 0, sizeof(s->fifo));
    qemu_set_irq(s->irq, 0);
}

static void imxrt_dac_realize(DeviceState *dev, Error **errp)
{
    IMXRTDACState *s = IMXRT_DAC(dev);
    SysBusDevice *sbd = SYS_BUS_DEVICE(dev);

    memory_region_init_io(&s->iomem, OBJECT(dev), &imxrt_dac_ops, s,
                          TYPE_IMXRT_DAC, IMXRT_DAC_MMIO_SIZE);
    sysbus_init_mmio(sbd, &s->iomem);
    sysbus_init_irq(sbd, &s->irq);
}

static const VMStateDescription vmstate_imxrt_dac = {
    .name = "imxrt.dac",
    .version_id = 1,
    .minimum_version_id = 1,
    .fields = (const VMStateField[]) {
        VMSTATE_UINT32(cr, IMXRTDACState),
        VMSTATE_UINT32(cr2, IMXRTDACState),
        VMSTATE_UINT32_ARRAY(fifo, IMXRTDACState, IMXRT_DAC_FIFO_DEPTH),
        VMSTATE_UINT32(wfp, IMXRTDACState),
        VMSTATE_UINT32(rfp, IMXRTDACState),
        VMSTATE_UINT32(count, IMXRTDACState),
        VMSTATE_UINT32(output, IMXRTDACState),
        VMSTATE_BOOL(udff, IMXRTDACState),
        VMSTATE_BOOL(ovff, IMXRTDACState),
        VMSTATE_END_OF_LIST()
    }
};

static void imxrt_dac_class_init(ObjectClass *klass, const void *data)
{
    DeviceClass *dc = DEVICE_CLASS(klass);
    ResettableClass *rc = RESETTABLE_CLASS(klass);

    dc->realize = imxrt_dac_realize;
    rc->phases.hold = imxrt_dac_reset_hold;
    dc->vmsd = &vmstate_imxrt_dac;
    dc->desc = "i.MX RT1170 DAC12";
}

static const TypeInfo imxrt_dac_info = {
    .name = TYPE_IMXRT_DAC,
    .parent = TYPE_SYS_BUS_DEVICE,
    .instance_size = sizeof(IMXRTDACState),
    .class_init = imxrt_dac_class_init,
};

static void imxrt_dac_register_types(void)
{
    type_register_static(&imxrt_dac_info);
}
type_init(imxrt_dac_register_types)
```
